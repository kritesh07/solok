        ServerAdmin $domain
        ServerName www.$domain
        DirectoryIndex index.php index.html
        DocumentRoot /var/www/html/$domain
</VirtualHost>" > /etc/apache2/sites-available/$domain.conf

        <Directory /var/www/html/server.com>\
